o3de_pal_dir(pal_dir ${CMAKE_CURRENT_LIST_DIR}/Platform/${PAL_PLATFORM_NAME} "${gem_restricted_path}" "${gem_path}" "${gem_parent_relative_path}")

ly_add_target(
    NAME MyMultiplayerGame.Static STATIC
    NAMESPACE Gem
    FILES_CMAKE
        mymultiplayergame_files.cmake
        mymultiplayergame_autogen_files.cmake
        ${pal_dir}/mymultiplayergame_${PAL_PLATFORM_NAME_LOWERCASE}_files.cmake
    INCLUDE_DIRECTORIES
        PRIVATE
            Source
            .
        PUBLIC
            Include
    BUILD_DEPENDENCIES
        PUBLIC
            AZ::AzGameFramework
            AZ::AzNetworking
            Gem::Multiplayer
            Gem::EMotionFXStaticLib
            Gem::PhysX
            Gem::StartingPointInput
            Gem::DebugDraw
            Gem::LyShine
        PRIVATE
            Gem::LmbrCentral.Static
            Gem::Multiplayer.Static
            Gem::PhysX.Static
            Gem::DebugDraw.Static
            Gem::ImGui.Static
            Gem::LyShine.Static
    AUTOGEN_RULES
        *.AutoComponent.xml,AutoComponent_Header.jinja,$path/$fileprefix.AutoComponent.h
        *.AutoComponent.xml,AutoComponent_Source.jinja,$path/$fileprefix.AutoComponent.cpp
        *.AutoComponent.xml,AutoComponentTypes_Header.jinja,$path/AutoComponentTypes.h
        *.AutoComponent.xml,AutoComponentTypes_Source.jinja,$path/AutoComponentTypes.cpp
)

ly_add_target(
    NAME MyMultiplayerGame ${PAL_TRAIT_MONOLITHIC_DRIVEN_MODULE_TYPE}
    NAMESPACE Gem
    FILES_CMAKE
        mymultiplayergame_shared_files.cmake
        ../../mymultiplayergame_asset_files.cmake
    INCLUDE_DIRECTORIES
        PRIVATE
            Source
            .
        PUBLIC
            Include
    BUILD_DEPENDENCIES
        PRIVATE
            Gem::MyMultiplayerGame.Static
            Gem::Atom_AtomBridge.Static
)

# if enabled, MyMultiplayerGame is used by all kinds of applications
ly_create_alias(NAME MyMultiplayerGame.Builders NAMESPACE Gem TARGETS Gem::MyMultiplayerGame)
ly_create_alias(NAME MyMultiplayerGame.Tools    NAMESPACE Gem TARGETS Gem::MyMultiplayerGame)
ly_create_alias(NAME MyMultiplayerGame.Clients  NAMESPACE Gem TARGETS Gem::MyMultiplayerGame)
ly_create_alias(NAME MyMultiplayerGame.Servers  NAMESPACE Gem TARGETS Gem::MyMultiplayerGame)

################################################################################
# Gem dependencies
################################################################################

# Maps the MyMultiplayerGame Project with the specified list of enabled gems
ly_enable_gems(PROJECT_NAME MyMultiplayerGame GEM_FILE enabled_gems.cmake)

# If we build a server, then add the project name to the list of server launcher projects
if(PAL_TRAIT_BUILD_SERVER_SUPPORTED)
    set_property(GLOBAL APPEND PROPERTY LY_LAUNCHER_SERVER_PROJECTS MyMultiplayerGame)
endif()

set_property(TARGET MyMultiplayerGame APPEND PROPERTY GAMELAUNCHER_ADDITIONAL_VS_DEBUGGER_COMMAND_ARGUMENTS "--console-command-file=\"client.cfg\"")
set_property(TARGET MyMultiplayerGame APPEND PROPERTY SERVERLAUNCHER_ADDITIONAL_VS_DEBUGGER_COMMAND_ARGUMENTS "--console-command-file=\"server.cfg\"")
